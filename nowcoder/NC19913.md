# 中位数图


特殊构造
```python
from collections import Counter
# import sys
# sys.stdin = open("in.txt", "r")

n, b = list(map(int, input().split()))
arr = list(map(int, input().split()))
p = -1
c = Counter()
ans = 1
for i, t in enumerate(arr):
    if t == b:
        p = i
        arr[i] = 0
    elif t < b:
        arr[i] = -1
    else:
        arr[i] = 1

s = 0
for i in range(p-1, -1, -1):
    s += arr[i]
    if s == 0:
        ans += 1
    c[s] += 1

s = 0
for i in range(p+1, n):
    s += arr[i]
    if s == 0:
        ans += 1
    if -s in c:
        ans += c[-s]

print(ans)
```
