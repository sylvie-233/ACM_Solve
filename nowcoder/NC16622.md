# 多项式输出


字符串构造模拟
```python
n = int(input())
arr = list(map(int, input().split()))
ans = ""
for i, t in enumerate(arr):
    if t > 0:
        if i != n:
            if i == n-1:
                if t != 1:
                    ans += f"+{t}x"
                else:
                    ans += f"+x"
            else:
                if t != 1:
                    ans += f"+{t}x^{n-i}"
                else:
                    ans += f"+x^{n-i}"
        else:
            ans += f"+{t}"
    elif t < 0:
        if i != n:
            if i == n-1:
                if t != -1:
                    ans += f"-{-t}x"
                else:
                    ans += f"-x"
            else:
                if t != -1:
                    ans += f"-{-t}x^{n-i}"
                else:
                    ans += f"-x^{n-i}"
        else:
            ans += f"-{-t}"
if ans[0] == "+":
    ans = ans[1:]
print(ans)
```
