# 字符串的展开


字符串构造模拟
```python
p1, p2, p3 = list(map(int, input().split()))
s = input()
n = len(s)
ans = ""
for i, c in enumerate(s):
    if c == "-":
        if i > 0 and i < n-1:
            l, r = s[i-1], s[i+1]
            if not ord(l) < ord(r):
                ans += '-'
                continue
            if l.isdigit() and r.isdigit():
                t = ""
                for i in range(ord(l)+1, ord(r)):
                    c = chr(i)
                    if p1 == 3:
                        c = "*"
                    c *= p2
                    if p3 == 1:
                        t = t + c
                    else:
                        t = c + t
                ans += t
            elif l.isalpha() and r.isalpha() and l.islower() and r.islower():
                t = ""
                for i in range(ord(l)+1, ord(r)):
                    c = chr(i)
                    if p1 == 1:
                        c = c.lower()
                    elif p1 == 2:
                        c = c.upper()
                    else:
                        c = "*"
                    c *= p2
                    if p3 == 1:
                        t = t + c
                    else:
                        t = c + t
                ans += t
            else:
                ans += '-'
        else:
            ans += "-"
    else:
        ans += c
        
print(ans)
```
