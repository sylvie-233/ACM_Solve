# 回文日期

字符串枚举
```python
start = input()
end = input()
start_int = int(start)
end_int = int(end)
start_year = int(start[:4])
end_year = int(end[:4])

day_dict = {
    "1": 31,
    "2": 28,
    "3": 31,
    "4": 30,
    "5": 31,
    "6": 30,
    "7": 31,
    "8": 31,
    "9": 30,
    "10": 31,
    "11": 30,
    "12": 31,
}

def is_valid(date: str) -> bool:
    year = int(date[:4])
    month = int(date[4:6])
    day = int(date[6:])
    if not str(month) in day_dict:
        return False
    is_leap_year = year % 400 == 0 or (year % 4 ==0 and year % 100 != 0)
    valid_days = day_dict[str(month)]
    if month == 2 and is_leap_year:
        valid_days = 29
    return 0 <= day and day <= valid_days

ans = 0
for i in range(start_year, end_year+1):
    str_year = str(i)
    caculate_date = str_year + str_year[::-1]
    if is_valid(caculate_date) and int(caculate_date) >= start_int and int(caculate_date) <= end_int:
        # print(caculate_date)
        ans += 1

print(ans)
```
