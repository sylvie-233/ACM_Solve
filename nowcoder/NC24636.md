NC16649-校门外的树 数据加强版，重叠区域计算
```c++
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <tuple>
#include <sstream>
#include <fstream>

using ll = long long;
using std::string;

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::ifstream fin("in.txt");
    std::istream &in = fin.is_open() ? fin : std::cin;
    int L, M, l, r;
    std::vector<std::tuple<int, int>> arr;
    in >> L >> M;
    for (int i = 0; i < M; i++) {
        in >> l >> r;
        arr.emplace_back(l, 0);
        arr.emplace_back(r, 1);
    }
    int sum = 0;
    int cnt = 0;
    int start = -1;
    std::ranges::sort(arr);
    for (int i = 0; i < 2 * M; i++) {
        int p = std::get<0>(arr[i]);
        int f = std::get<1>(arr[i]);
        if (start == -1) {
            start = p;
        }
        if (f == 0) {
            cnt++;
        } else {
            cnt--;
            if (cnt == 0) {
                sum += p - start + 1;
                start = -1;
            }
        }
    }
    std::cout << (1 + L - sum) << std::endl;
    return 0;
}
```
