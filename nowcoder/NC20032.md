# 激光炸弹


二维前缀和
```c++
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <tuple>
#include <sstream>
#include <fstream>

using ll = long long;
using std::string;

class PrefixSum2D {
    private:
        int n, m;
        std::vector<std::vector<int>> a;   // 原矩阵 (1-based)
        std::vector<std::vector<int>> ps;  // 前缀和矩阵 (1-based)

    public:
        // 构造函数
        PrefixSum2D(int _n, int _m) : n(_n), m(_m) {
            a.assign(n + 1, std::vector<int>(m + 1, 0));
            ps.assign(n + 1, std::vector<int>(m + 1, 0));
        }

        // 从外部矩阵构建（矩阵下标为 1-based；若是 0-based，请先偏移）
        void initFromMatrix(const std::vector<std::vector<int>>& mat) {
            for (int i = 1; i <= n; i++) {
                for (int j = 1; j <= m; j++) {
                    a[i][j] = mat[i][j];
                }
            }
        }

        // 单点加入 —— 若你逐个读输入，可直接 add(i,j,val)
        void add(int x, int y, int v) {
            a[x][y] += v;
        }

        // 构建前缀和（必须在全部 add() 完成后调用一次）
        void build() {
            for (int i = 1; i <= n; i++) {
                int row_sum = 0;
                for (int j = 1; j <= m; j++) {
                    row_sum += a[i][j];
                    ps[i][j] = ps[i-1][j] + row_sum;
                }
            }
        }

        // 查询 (x1, y1) 到 (x2, y2) 的子矩形和
        int query(int x1, int y1, int x2, int y2) {
            if (x1 > x2) std::swap(x1, x2);
            if (y1 > y2) std::swap(y1, y2);
            return ps[x2][y2]
                - ps[x1-1][y2]
                - ps[x2][y1-1]
                + ps[x1-1][y1-1];
        }
    };

int main() {
    std::ifstream fin("in.txt");
    std::istream &in = fin.is_open() ? fin : std::cin;

    int n, R;
    in >> n >> R;

    PrefixSum2D ps(5001, 5001);
    int x, y, v;
    for (int i = 0; i < n; i++) {
        in >> x >> y >> v;
        ps.add(x + 1, y + 1, v);
    }
    ps.build();
    int ans = 0;
    for (int i = 1; i <= 5001; i++) {
        if (i + R - 1 > 5001) break;
        for (int j = 1; j <= 5001; j++) {
            if (j + R - 1 > 5001) break;
            ans = std::max(ans, ps.query(i, j, i + R - 1, j + R - 1));
        }
    }
    std::cout << ans << std::endl;
    return 0;
}
```
