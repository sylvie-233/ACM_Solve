二分查找

```python
def binary_search(nums: list[int], target: int) -> int:
    left = 0
    right = len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] >= target:
            right = mid - 1
        else:
            left = mid + 1
        
    return left if (0 <= left <= len(nums) - 1) and nums[left] == target else -1

if __name__ == "__main__":
    n, m = list(map(int, input().split()))
    nums = list(map(int, input().split()))
    queries = list(map(int, input().split()))
    # print(n, m)
    # print(nums, queries)
    res = []
    for q in queries:
        idx = binary_search(nums, q)
        # print(idx)
        if idx == -1:
            res.append(-1)
        else:
            res.append(idx + 1)
    print(" ".join(list(map(str, res))))
```
