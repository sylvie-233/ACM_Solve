# 快速幂 


快速幂模板题
```c++
#include <iostream>
#include <format>

using ll = long long;

ll qpow(ll a, ll b, ll mod) {
    ll res = 1;
    a %= mod;

    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

int main() {
    ll a, b, q;
    std::cin >> a >> b >> q;
    std::cout << std::format("{}^{} mod {}={}", a, b, q, qpow(a, b, q)) << std::endl;
    return 0;
}
```
